# Script code with comments

```
# ============================================================================
# Часть I: Получение данных через WMI (Windows Management Instrumentation)
# ============================================================================

# Получаем имя компьютера из класса Win32_ComputerSystem
# Get-WmiObject - устаревший, но все еще работающий командлет для запросов WMI
# -Query "SELECT Name FROM Win32_ComputerSystem" - WQL запрос (WMI Query Language)
#   SELECT - ключевое слово SQL для выборки данных
#   Name - имя поля (свойства), которое хотим получить
#   FROM - ключевое слово SQL, указывающее источник данных
#   Win32_ComputerSystem - класс WMI, содержащий информацию о компьютерной системе
# .Name - обращаемся к свойству Name объекта, возвращаемого Get-WmiObject
# ( ... ) - круглые скобки группируют выражение, чтобы можно было обратиться к свойству
$computerName = (Get-WmiObject -Query "SELECT Name FROM Win32_ComputerSystem").Name

# Получаем список пользователей из класса Win32_UserAccount
# Win32_UserAccount - класс WMI, содержащий информацию об учетных записях пользователей
# Без WHERE - выбираются все пользователи в системе
$users = Get-WmiObject -Query "SELECT Name FROM Win32_UserAccount"

# Получаем сетевые адаптеры и их MAC-адреса
# Win32_NetworkAdapter - класс WMI, содержащий информацию о сетевых адаптерах
# WHERE MACAddress IS NOT NULL - условие отбора (фильтрация)
#   WHERE - ключевое слово SQL для фильтрации записей
#   MACAddress - имя поля (свойства) с MAC-адресом
#   IS NOT NULL - условие, отбирающее только записи, где MAC-адрес существует
#   NULL - специальное значение, означающее "пусто" или "не определено"
$networkAdapters = Get-WmiObject -Query "SELECT Name, MACAddress FROM Win32_NetworkAdapter WHERE MACAddress IS NOT NULL"

# Выводим заголовок "Computer name:" на экран
# Write-Host - командлет для вывода текста в консоль
Write-Host "Computer name:"

# Выводим значение переменной $computerName (имя компьютера)
Write-Host $computerName

# Выводим пустую строку для разделения блоков информации
# "" - пустая строка (два символа двойной кавычки без содержимого)
Write-Host ""

# Выводим заголовок "List of users:"
Write-Host "List of users:"

# Обрабатываем коллекцию пользователей
# $users | ForEach-Object { ... } - передаем объекты из $users по конвейеру (|)
#   | - символ конвейера (pipe), передает вывод левой команды на вход правой
# ForEach-Object - командлет для выполнения блока кода для каждого объекта во входных данных
# { ... } - блок скрипта (script block), выполняется для каждого пользователя
#   $_ - автоматическая переменная, представляющая текущий объект в конвейере
#   .Name - обращаемся к свойству Name текущего объекта пользователя
$users | ForEach-Object {
    Write-Host $_.Name
}

# Выводим пустую строку
Write-Host ""

# Выводим заголовок "Network cards and MAC addresses:"
Write-Host "Network cards and MAC addresses:"

# Обрабатываем коллекцию сетевых адаптеров
# $networkAdapters | ForEach-Object - передаем каждый адаптер в блок обработки
$networkAdapters | ForEach-Object {
    # Выводим информацию об адаптере в формате "Имя - MAC-адрес"
    # "$(...)" - строковая интерполяция (подстановка значений переменных)
    #   $() - подстановочное выражение, вычисляет значение внутри скобок и вставляет в строку
    #   $_.Name - имя текущего адаптера
    #   $_.MACAddress - MAC-адрес текущего адаптера
    Write-Host "$($_.Name) - $($_.MACAddress)"
}

# ============================================================================
# Часть II: Получение программ автозагрузки через WMI
# ============================================================================

# Получаем список программ в автозагрузке из класса Win32_StartupCommand
# Win32_StartupCommand - класс WMI, содержащий информацию о программах,
#   запускаемых автоматически при старте системы или входе пользователя
# SELECT Name, Command, Location - выбираем имя программы, команду запуска и расположение в реестре
$startupPrograms = Get-WmiObject -Query "SELECT Name, Command, Location FROM Win32_StartupCommand"

# ============================================================================
# Часть III: Сохранение данных в файл
# ============================================================================

# Задаем каталог для сохранения отчета
# "C:\SysAdmLab2" - строковый литерал с путем к каталогу
$reportDir = "C:\SysAdmLab2"

# Формируем полный путь к файлу отчета
# Join-Path - командлет для объединения пути и имени файла с правильными разделителями
#   $reportDir - переменная с путем к каталогу
#   "$computerName.rep" - имя файла, составленное из имени компьютера и расширения .rep
#     $computerName - переменная с именем компьютера
#     .rep - расширение файла (точка + rep)
$reportFile = Join-Path $reportDir "$computerName.rep"

# Формируем массив строк для записи в файл
# @( ... ) - оператор создания массива (array subexpression operator)
#   Создает массив, содержащий все элементы, перечисленные внутри скобок
$fileContent = @(
    # Первая строка - секция пользователей
    # "[users]" - строковый литерал, обозначающий начало секции в формате INI-подобного файла
    "[users]"
    
    # Получаем имена пользователей, преобразуя каждый объект в его имя
    # ($users | ForEach-Object { $_.Name }) - выражение, возвращающее массив имен
    #   Каждое имя станет отдельным элементом массива
    ($users | ForEach-Object { $_.Name })
    
    # Пустая строка для разделения секций
    ""
    
    # Секция сетевых адаптеров
    "[network]"
    
    # Формируем строки для каждого адаптера в формате "Имя, MAC-адрес"
    # "$($_.Name), $($_.MACAddress)" - интерполированная строка с именем и MAC-адресом
    #   Запятая и пробел - разделители, требуемые по заданию
    ($networkAdapters | ForEach-Object { "$($_.Name), $($_.MACAddress)" })
    
    # Пустая строка
    ""
    
    # Секция программ автозагрузки
    "[startup]"
    
    # Формируем строки для каждой программы автозагрузки в формате "Name: ..., Command: ..., Location: ..."
    # "Name: $($_.Name), Command: $($_.Command), Location: $($_.Location)" - строка с тремя полями
    #   Name: - текстовый идентификатор поля
    #   $($_.Name) - имя программы
    #   Command: - идентификатор поля команды
    #   $($_.Command) - команда запуска (путь к исполняемому файлу с параметрами)
    #   Location: - идентификатор поля расположения
    #   $($_.Location) - путь в реестре, где зарегистрирована автозагрузка
    ($startupPrograms | ForEach-Object { "Name: $($_.Name), Command: $($_.Command), Location: $($_.Location)" })
)

# Сохраняем массив строк в файл
# $fileContent | Out-File - передаем содержимое массива по конвейеру в командлет Out-File
# Out-File - командлет для записи вывода в файл
#   -FilePath $reportFile - параметр, указывающий путь к файлу
#     -FilePath - имя параметра (можно сократить до -Path)
#     $reportFile - переменная с полным путем к файлу
#   -Encoding utf8 - параметр, задающий кодировку файла
#     utf8 - кодировка UTF-8 без BOM (Byte Order Mark)
#     Позволяет корректно сохранять русские символы и другие не-ASCII символы
$fileContent | Out-File -FilePath $reportFile -Encoding utf8

# Выводим пустую строку для отделения от предыдущего вывода
Write-Host ""

# Выводим сообщение о сохранении данных
Write-Host "Data saved to file:"

# Выводим полный путь к созданному файлу отчета
Write-Host $reportFile
```
